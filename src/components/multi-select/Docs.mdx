# MultiSelect

A combination of generic [useMultiSelectFieldProps](?path=/docs/hooks-usemultiselectfieldprops--docs) and [useSelectOptions](?path=/docs/hooks-useselectoptions--docs)
which renders the options as the children of `<select multiple />`.

```ts
import { MultiSelect } from "@form-atoms/field";
```

## Example

```tsx
import {
  Select,
  MultiSelectProps,
  FieldLabel,
  FieldErrors,
  ZodArrayField,
} from "@form-atoms/field";

const SelectField = <Option, Field extends ZodArrayField>({
  field,
  label,
  ...props
}: {
  label: ReactNode;
} & MultiSelectProps<Option, Field>) => (
  <div style={{ margin: "20px 0" }}>
    <FieldLabel field={field} label={label} />
    <MultiSelect {...props} field={field} />
    <FieldErrors field={field} />
  </div>
);
```
